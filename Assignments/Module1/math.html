\begin{align*} \text{cov}[x_i, x_j] &= \mathbb{E} [(x_i - \mathbb{E}[x_i])(x_j - \mathbb{E}[x_j])]\\ &=\mathbb{E}\left[(x_i - \mathbb{E}[x_i])\left\{\sum_{k \in \text{pa}_j}\,w_{jk}(x_k - \mathbb{E}[x_k])+\sqrt{v_j}\epsilon_j\right\} \right ]\\ &=\sum_{k \in \text{pa}_j}\,w_{jk}\,\text{cov}[x_i, x_j] + I_{ij}v_j \end{align*}
